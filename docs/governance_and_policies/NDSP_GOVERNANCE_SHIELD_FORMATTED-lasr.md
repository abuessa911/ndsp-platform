# 21) قالب السكربت القياسي المعتمد

استخدم هذا القالب لأي مهمة مستقبلية:

```bash
#!/usr/bin/env bash
set -Eeuo pipefail

TASK_NAME="NDSP_TASK_NAME"
STAMP="$(date +%Y%m%d_%H%M%S)"

ROOT="${ROOT:-/home/nawaf511/empire-core-new}"
BACKEND="$ROOT/backend"
APP="$BACKEND/app"

REPORT_DIR="/home/nawaf511/ndsp_launch_reports"
BACKUP_DIR="/home/nawaf511/ndsp_backups/${TASK_NAME}_${STAMP}"
REPORT="$REPORT_DIR/${TASK_NAME}_${STAMP}.md"

PYTHON_BIN="$BACKEND/venv/bin/python"
if [ ! -x "$PYTHON_BIN" ]; then
  PYTHON_BIN="python3"
fi

mkdir -p "$REPORT_DIR" "$BACKUP_DIR"

log() {
  echo "$@" | tee -a "$REPORT"
}

section() {
  echo "" | tee -a "$REPORT"
  echo "$1" | tee -a "$REPORT"
  echo "------------------------------------------------------------" | tee -a "$REPORT"
}

{
  echo "# $TASK_NAME"
  echo ""
  echo "- STAMP=$STAMP"
  echo "- ROOT=$ROOT"
  echo "- BACKEND=$BACKEND"
  echo "- BACKUP_DIR=$BACKUP_DIR"
  echo "- REPORT=$REPORT"
  echo "- PYTHON_BIN=$PYTHON_BIN"
} | tee "$REPORT"

section "1) Preconditions"

if [ ! -d "$ROOT" ]; then
  log "ERROR: ROOT not found: $ROOT"
  exit 1
fi

if [ ! -d "$BACKEND" ]; then
  log "ERROR: BACKEND not found: $BACKEND"
  exit 1
fi

log "PRECONDITIONS_OK=True"

section "2) Backup"

# Example:
# TARGET_FILE="$APP/core/example.py"
# if [ -f "$TARGET_FILE" ]; then
#   mkdir -p "$BACKUP_DIR/app/core"
#   cp -a "$TARGET_FILE" "$BACKUP_DIR/app/core/example.py.bak"
#   log "BACKUP: $TARGET_FILE"
# fi

log "BACKUP_DIR=$BACKUP_DIR"
log "BACKUP_OK=True"

section "3) Patch / Write / Update"

# Write your patch here.
# Always log created/updated files.

log "PATCH_OK=True"

section "4) Compile / Syntax Check"

# Example Python compile:
# PYTHONPATH="$ROOT" "$PYTHON_BIN" -m py_compile "$APP/core/example.py"

log "COMPILE_OK=True"

section "5) Test / Verification"

# Add test commands here.

log "TEST_OK=True"

section "6) Optional Restart"

# Example:
# sudo systemctl restart ndsp-api.service
# sudo systemctl status ndsp-api.service --no-pager -l | head -40 | tee -a "$REPORT"

log "RESTART_SKIPPED_OR_OK=True"

section "7) Smoke Test"

# Example:
# CODE="$(curl -sk -o /tmp/ndsp_smoke.json -w '%{http_code}' http://127.0.0.1:9001/health || true)"
# log "HTTP_CODE=$CODE"

log "SMOKE_OK=True"

section "8) Final Assertions"

ASSERT_OK=True

if [ "$ASSERT_OK" = "True" ]; then
  log "ASSERT_OK=True"
  log "FINAL_STATUS=${TASK_NAME}_DONE"
else
  log "ASSERT_OK=False"
  log "FINAL_STATUS=${TASK_NAME}_FAILED"
  exit 1
fi

log "REPORT=$REPORT"
log "BACKUP_DIR=$BACKUP_DIR"
log "ROLLBACK_HINT=Restore changed files from BACKUP_DIR if needed"
log "DONE"
```
